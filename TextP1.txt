file = open('data.txt', 'r')
text = []
for line in file:
  text.append(str(line))

punc = '''â€¢123456789!()-[]{};:'"\,<>./?@#$%^&*_~'''

new_text = []
n = len(text)
for i in range(n):
  line = ""
  for j in text[i]:
    if j not in punc:
      line += j
  new_text.append(line)

for line in new_text:
  print(line)

words = []
for i in text:
  words += i.split()

#taking counting the number of words in data file
len(words)
